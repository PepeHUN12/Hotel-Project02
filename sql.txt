CREATE TABLE rooms (
    ID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    Name varchar(255),
    Price int,
    Size int,
    Bed varchar(255),
    Adult int,
    Child int,
    Services varchar(255),
    Availability int);
